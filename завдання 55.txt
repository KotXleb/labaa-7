#include <iostream>

using namespace std;

int main() {
    int M, N;
    cin >> M >> N;

    int a[100][100];
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++) {
            cin >> a[i][j];
        }
    }
    int half = M / 2;
    for (int i = 0; i < half; i++) {
        for (int j = 0; j < N; j++) {
            int temp = a[i][j];
            a[i][j] = a[i + half][j];
            a[i + half][j] = temp;
        }
    }
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++) {
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
