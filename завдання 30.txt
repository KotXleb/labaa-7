#include <iostream>
using namespace std;

int main() {
    setlocale(LC_ALL, "Ukr");
    int M, N;
    cout << "Введіть кількість рядків M та стовпців N:" << endl;
    cin >> M >> N;
    double a[100][100]; // Масив із запасом

    // 1. Читаємо всю матрицю
    for (int i = 0; i < M; i++)
        for (int j = 0; j < N; j++)
            cin >> a[i][j];

    // 2. Рахуємо по стовпцях
    for (int j = 0; j < N; j++) {
        double sum = 0;
        for (int i = 0; i < M; i++) sum += a[i][j];

        double avg = sum / M;
        int count = 0;

        for (int i = 0; i < M; i++) {
            if (a[i][j] > avg) count++;
        }
        cout << count << " ";
    }

    return 0;
}
